O Projekcie

Pixel Adventure to prosta gra platformowa stworzona w JavaScript, wykorzystująca HTML5 Canvas do renderowania grafiki. Inspirowana klasykami gatunku jak Super Mario Bros czy Sonic the Hedgehog, oferuje nostalgiczną rozgrywkę w stylu retro z nowoczesnym twistem.

W grze sterujemy bohaterem o imieniu Pixie, który musi przemierzyć różnorodne poziomy pełne platform, przeszkód i wrogów. Celem jest zebranie wszystkich klejnotów na każdym poziomie i dotarcie do portalu, który przenosi do następnego etapu.

  • Responsywność - gra dostosowuje się do różnych rozmiarów ekranu
  • Zapisywanie postępu w LocalStorage przeglądarki

Proces Tworzenia

Projekt Pixel Adventure powstał jako rozwinięcie mojej pierwszej prostej gry, którą stworzyłem w Scratch kilka lat temu. Chciałem sprawdzić, czy jestem w stanie przenieść podobną koncepcję na "prawdziwy" język programowania i dodać więcej funkcjonalności.

Projekt rozpocząłem od zaprojektowania bohatera i podstawowych elementów otoczenia w Aseprite. Następnie zaimplementowałem podstawową mechanikę poruszania się i kolizji. Gdy te działały poprawnie, dodawałem kolejne elementy: wrogów, zbieralne przedmioty, różne typy platform.

Największe wyzwania

Największym wyzwaniem było zaimplementowanie poprawnego systemu kolizji, który działałby ze wszystkimi elementami otoczenia. Szczególnie trudne było zapewnienie, aby postać "stała" na platformach, a nie przenikała przez nie lub odbijała się w niekontrolowany sposób.

Drugim wyzwaniem było zoptymalizowanie renderowania gry tak, aby działała płynnie nawet na słabszych urządzeniach. Zastosowałem technikę renderowania warstwowego i ograniczenie odświeżania do elementów widocznych na ekranie.

System kolizji


// Klasa odpowiedzialna za wykrywanie kolizji
class CollisionDetector {
    constructor(game) {
        this.game = game;
    }
    
    // Sprawdza kolizje postaci z platformami
    checkPlatformCollisions(character, platforms) {
        const characterRect = {
            x: character.x,
            y: character.y,
            width: character.width,
            height: character.height
        };
        
        let onGround = false;
        let hitCeiling = false;
        let hitWall = false;
        
        platforms.forEach(platform => {
            // Pomijamy platformy poza ekranem
            if (!this.isInView(platform)) return;
            
            const platformRect = {
                x: platform.x,
                y: platform.y,
                width: platform.width,
                height: platform.height
            };
            
            if (this.checkRectCollision(characterRect, platformRect)) {
                // Określamy kierunek kolizji na podstawie pozycji
                const overlapX = Math.min(
                    character.x + character.width - platform.x,
                    platform.x + platform.width - character.x
                );
                
                const overlapY = Math.min(
                    character.y + character.height - platform.y,
                    platform.y + platform.height - character.y
                );
                
                // Kolizja nastąpiła w osi z mniejszym nakładaniem się
                if (overlapX < overlapY) {
                    // Kolizja pozioma (ściana)
                    if (character.x < platform.x) {
                        character.x = platform.x - character.width;
                    } else {
                        character.x = platform.x + platform.width;
                    }
                    character.velocityX = 0;
                    hitWall = true;
                } else {
                    // Kolizja pionowa (podłoga lub sufit)
                    if (character.y < platform.y) {
                        character.y = platform.y - character.height;
                        character.velocityY = 0;
                        onGround = true;
                    } else {
                        character.y = platform.y + platform.height;
                        character.velocityY = 0;
                        hitCeiling = true;
                    }
                }
                
                // Specjalne zachowanie dla różnych typów platform
                if (platform.type === 'bounce' && onGround) {
                    character.velocityY = -character.jumpForce * 1.5;
                    this.game.audio.playSound('bounce');
                } else if (platform.type === 'moving') {
                    // Postać porusza się wraz z platformą
                    character.x += platform.velocityX;
                }
            }
        });
        
        return { onGround, hitCeiling, hitWall };
    }
    
    // Sprawdza kolizję dwóch prostokątów
    checkRectCollision(rect1, rect2) {
        return (
            rect1.x < rect2.x + rect2.width &&
            rect1.x + rect1.width > rect2.x &&
            rect1.y < rect2.y + rect2.height &&
            rect1.y + rect1.height > rect2.y
        );
    }
    
    // Sprawdza, czy obiekt jest widoczny na ekranie
    isInView(object) {
        return (
            object.x + object.width > this.game.camera.x &&
            object.x < this.game.camera.x + this.game.width &&
            object.y + object.height > this.game.camera.y &&
            object.y < this.game.camera.y + this.game.height
        );
    }
}

// Klasa zarządzająca animacjami postaci
class CharacterAnimator {
    constructor(character) {
        this.character = character;
        this.animations = {
            idle: {
                frames: 4,
                frameSpeed: 10,
                spriteSheet: document.getElementById('idle-spritesheet')
            },
            run: {
                frames: 6,
                frameSpeed: 8,
                spriteSheet: document.getElementById('run-spritesheet')
            },
            jump: {
                frames: 2,
                frameSpeed: 6,
                spriteSheet: document.getElementById('jump-spritesheet')
            },
            fall: {
                frames: 2,
                frameSpeed: 6,
                spriteSheet: document.getElementById('fall-spritesheet')
            }
        };
        
        this.currentAnimation = 'idle';
        this.frameIndex = 0;
        this.frameCounter = 0;
        this.frameWidth = 32;
        this.frameHeight = 32;
        this.facingRight = true;
    }
    
    // Aktualizuje stan animacji
    update() {
        // Określenie odpowiedniej animacji na podstawie stanu postaci
        let newAnimation = 'idle';
        
        if (!this.character.onGround) {
            newAnimation = this.character.velocityY < 0 ? 'jump' : 'fall';
        } else if (Math.abs(this.character.velocityX) > 0.1) {
            newAnimation = 'run';
        }
        
        // Zmiana kierunku postaci
        if (this.character.velocityX > 0) {
            this.facingRight = true;
        } else if (this.character.velocityX < 0) {
            this.facingRight = false;
        }
        
        // Resetuj licznik klatek przy zmianie animacji
        if (this.currentAnimation !== newAnimation) {
            this.frameIndex = 0;
            this.frameCounter = 0;
            this.currentAnimation = newAnimation;
        }
        
        // Aktualizacja klatki animacji
        this.frameCounter++;
        const { frameSpeed } = this.animations[this.currentAnimation];
        
        if (this.frameCounter >= frameSpeed) {
            this.frameCounter = 0;
            this.frameIndex = (this.frameIndex + 1) % this.animations[this.currentAnimation].frames;
        }
    }
    
    // Rysuje aktualną klatkę animacji
    draw(ctx) {
        const animation = this.animations[this.currentAnimation];
        const spriteSheet = animation.spriteSheet;
        
        ctx.save();
        
        if (!this.facingRight) {
            // Odbij poziomo dla postaci skierowanej w lewo
            ctx.translate(this.character.x + this.character.width, this.character.y);
            ctx.scale(-1, 1);
            ctx.drawImage(
                spriteSheet,
                this.frameIndex * this.frameWidth, 0,
                this.frameWidth, this.frameHeight,
                0, 0,
                this.character.width, this.character.height
            );
        } else {
            ctx.drawImage(
                spriteSheet,
                this.frameIndex * this.frameWidth, 0,
                this.frameWidth, this.frameHeight,
                this.character.x, this.character.y,
                this.character.width, this.character.height
            );
        }
        
        ctx.restore();
    }
}

System dźwięku


// Klasa zarządzająca dźwiękami i muzyką
class AudioManager {
    constructor() {
        this.audioContext = new (window.AudioContext || window.webkitAudioContext)();
        this.sounds = {};
        this.music = {};
        this.musicVolume = 0.5;
        this.soundVolume = 0.8;
        this.currentMusic = null;
        
        // Załaduj dźwięki
        this.loadSounds();
    }
    
    // Ładuje wszystkie dźwięki i muzykę
    async loadSounds() {
        const soundFiles = {
            jump: 'sounds/jump.wav',
            coin: 'sounds/coin.wav',
            hurt: 'sounds/hurt.wav',
            powerup: 'sounds/powerup.wav',
            bounce: 'sounds/bounce.wav',
            victory: 'sounds/victory.wav',
            gameOver: 'sounds/game-over.wav'
        };
        
        const musicFiles = {
            theme1: 'music/theme1.mp3',
            theme2: 'music/theme2.mp3',
            theme3: 'music/theme3.mp3',
            boss: 'music/boss.mp3',
            menu: 'music/menu.mp3'
        };
        
        // Ładuje dźwięki
        for (const [name, path] of Object.entries(soundFiles)) {
            try {
                const response = await fetch(path);
                const arrayBuffer = await response.arrayBuffer();
                const audioBuffer = await this.audioContext.decodeAudioData(arrayBuffer);
                this.sounds[name] = audioBuffer;
            } catch (error) {
                console.error(`Nie udało się załadować dźwięku: ${path}`, error);
            }
        }
        
        // Ładuje muzykę
        for (const [name, path] of Object.entries(musicFiles)) {
            try {
                const response = await fetch(path);
                const arrayBuffer = await response.arrayBuffer();
                const audioBuffer = await this.audioContext.decodeAudioData(arrayBuffer);
                this.music[name] = audioBuffer;
            } catch (error) {
                console.error(`Nie udało się załadować muzyki: ${path}`, error);
            }
        }
    }
    
    // Odtwarza dźwięk
    playSound(name) {
        if (!this.sounds[name]) return;
        
        const source = this.audioContext.createBufferSource();
        source.buffer = this.sounds[name];
        
        const gainNode = this.audioContext.createGain();
        gainNode.gain.value = this.soundVolume;
        
        source.connect(gainNode);
        gainNode.connect(this.audioContext.destination);
        
        source.start();
    }
    
    // Odtwarza muzykę w pętli
    playMusic(name) {
        // Zatrzymaj aktualną muzykę
        if (this.currentMusic) {
            this.currentMusic.stop();
            this.currentMusic = null;
        }
        
        if (!this.music[name]) return;
        
        const source = this.audioContext.createBufferSource();
        source.buffer = this.music[name];
        source.loop = true;
        
        const gainNode = this.audioContext.createGain();
        gainNode.gain.value = this.musicVolume;
        
        source.connect(gainNode);
        gainNode.connect(this.audioContext.destination);
        
        source.start();
        this.currentMusic = source;
    }
    
    // Zatrzymuje muzykę
    stopMusic() {
        if (this.currentMusic) {
            this.currentMusic.stop();
            this.currentMusic = null;
        }
    }
    
    // Ustawia głośność muzyki
    setMusicVolume(volume) {
        this.musicVolume = Math.max(0, Math.min(1, volume));
        // Aktualizacja głośności aktualnie odtwarzanej muzyki
        // ...
    }
    
    // Ustawia głośność dźwięków
    setSoundVolume(volume) {
        this.soundVolume = Math.max(0, Math.min(1, volume));
    }
}
